# ── OpenCV — HEADLESS MUST COME FIRST ─────────────────────────────────────────
# Pin opencv-python-headless before mediapipe is listed so pip's dependency
# resolver sees the headless variant first and does not install opencv-python
# (the GUI build) alongside it.
#
# Root cause of "libGL.so.1: cannot open shared object file":
#   mediapipe>=0.10.11 lists `opencv-python` as a dependency in its own
#   metadata. When pip resolves both packages together, it installs the GUI
#   build which links against libGL — absent in Streamlit Cloud containers.
#
# Fix: pin opencv-python-headless here first, then pin mediapipe to a version
#   whose metadata accepts the headless variant.
opencv-python-headless==4.8.1.78

# ── MediaPipe ─────────────────────────────────────────────────────────────────
# 0.10.9 is the last release whose wheel metadata does NOT hard-require
# opencv-python (GUI). Later versions re-introduced that hard dependency.
mediapipe==0.10.9

# ── NumPy ─────────────────────────────────────────────────────────────────────
# NumPy 2.x introduced breaking changes for some mediapipe internals.
# Pin to the last stable 1.x series.
numpy==1.26.4

# ── Streamlit + WebRTC ────────────────────────────────────────────────────────
streamlit==1.35.0
streamlit-webrtc==0.47.9
aiortc==1.9.0
av==12.3.0

# ── PDF export ────────────────────────────────────────────────────────────────
reportlab==4.2.5
